% Poor man's version control which allows us to easily extract a stable snapshot.
% 1: stable branch, 0: latest branch
\def\stable{1}

% You can use this macro to keep two versions of a text around:
%   \stableLatest{old text that's stable}{new text that's not stable but you want to see by default}
\if\stable1
  \newcommand\stableLatest[2]{#1} % Stable
\else
  \newcommand\stableLatest[2]{#2} % Latest
\fi
